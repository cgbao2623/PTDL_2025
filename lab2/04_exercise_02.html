""" CÁC BẠN LÀM BÀI Ở ĐÂY """
# Khai báo tham số phân phối chuẩn
mu = 1.0        # trung bình
sigma = 1.0     # độ lệch chuẩn

# Tạo dãy x trong khoảng [mu - 5σ, mu + 5σ]
x_min = mu - 5 * sigma
x_max = mu + 5 * sigma
x = np.linspace(x_min, x_max, 200)  # 200 điểm đều nhau trong khoảng

# Tính y = hàm mật độ xác suất chuẩn
y = (1 / (sigma * np.sqrt(2 * np.pi))) * np.exp(-((x - mu) ** 2) / (2 * sigma ** 2))

print("b. Sinh cặp (x, y) với y làm hàm phân phối chuẩn")
print(f"+ x: {x_min} <= min [{np.min(x): .2f}] <= max [{np.max(x): .2f}] <= {x_max}")
print(f"+ Ve ham so y(x)")
plt.figure(figsize=(6,6))      # kích thước bản vẽ
plt.scatter(x, y, s=10, c='blue', alpha=0.6, label="PDF N(1,1)")
plt.title("Hàm phân phối chuẩn N(μ=1, σ²=1)")
plt.xlabel("x")
plt.ylabel("y")
plt.grid(alpha=0.3)
plt.legend()
plt.savefig("4b.png")          # lưu đồ thị
plt.show()                     # hiển thị đồ thị

Giải hệ phương trình tuyến tính sau: $$ \left\{\begin{matrix} 4x_1 + 3x_2 - 5x_3 = 2 \\ -2x_1 - 4x_2 + 5x_3 = 5 \\ 8x_1 + 8x_2 = -3 \\ \end{matrix}\right. $$

""" CÁC BẠN LÀM BÀI Ở ĐÂY """
# ...
# Tạo ma trận hệ số A và vector hằng số y
A = np.array([[4, 3, -5],
              [-2, -4, 5],
              [8, 8, 0]])
y = np.array([2, 5, -3])

# Giải hệ phương trình Ax = y
x = np.linalg.solve(A, y)

# Kiểm tra nghiệm bằng cách tính lại Ax
yy = A @ x
diff = np.linalg.norm(yy - y, 2)

print(f"+ Ma tran A: \n{A}")
print(f"+ Vector y: {y}")
print(f"+ Nghiem x: {x}")
print(f"+ Kiem tra: yy = Ax = {yy} ==> ||yy -y ||_2 = {diff: .2f}")

Bài 3. Tính gần đúng số pi dùng phương pháp Monte Carlo¶

Vẽ một đường tròn $C$ có bán kính $r$ nội tiếp một hình vuông $C$, và đặt ngẫu nhiên $N$ chấm lên hình vuông. Tỉ lệ các chấm nằm trong hình tròn trên tổng số chấm xấp xỉ bằng diện tích của hình tròn chia cho hình vuông như sau: $$ \frac{N_C}{N} \approx \frac{\mathbb{S}_C}{\mathbb{S}_N} $$ nơi mà $N_C$ là các chấm trong hình tròn, $N$ là tổng số chấm, $\mathbb{S}_C$ và $\mathbb{S}_N$ lần lượt là diện tích của hình tròn và hình vuông. $$ \mathbb{S}_C = \pi * r^2 $$ $$ \mathbb{S}_N = (2*r)^2 = 4r^2 $$ Do đó, công thức trên trở thành: $$ \frac{N_C}{N} \approx \frac{\pi * r^2}{4r^2} = \frac{\pi}{4} $$ Suy ra, ta có giá trị số $\pi$ được tính như sau: $$ \pi \approx \frac{4N_C}{N} $$

Các bạn hãy lập trình tính sấp xỉ số $\pi$ dùng phương pháp Monte Carlo. Cho biết các sai số với $N = 100$, $N = 10000$, và $N = 1000000$. Vẽ đồ thị minh họa cho chương trình.

Hướng dẫn: Viết chương trình phát sinh $N$ điểm $(x_i, y_i)$ với $x_i, y_i \in [-1, -1]$. $(x_i, y_i)$ thuộc hình tròn tâm $(0, 0)$ bán kính 1 nội tiếp hình vuông có tâm tại $(0, 0)$ độ dài cạnh 2 khi và chỉ khi $x_i^2 + y_i^2 <= 1$.

import numpy as np

def calc_pi_monte_carlo(n = 100):
    pi = 0
    
    # gợi ý: sử dụng np.random.rand(s1, s2, ...) --> sinh ngẫu nhiên các số trong nửa đoạn [0.0, 1.0)
    """ CÁC BẠN LÀM BÀI Ở ĐÂY """    
    # ...
    # Sinh ngẫu nhiên n điểm (x, y) trong [-1, 1] x [-1, 1]
    x = np.random.rand(n) * 2 - 1  # scale [0,1) -> [-1,1)
    y = np.random.rand(n) * 2 - 1
    # Kiểm tra số điểm nằm trong đường tròn x^2 + y^2 <= 1
    inside = x**2 + y**2 <= 1
    count_inside = np.sum(inside)
    # Công thức xấp xỉ pi
    pi = 4 * count_inside / n
    
    return pi
